Patrol 动作 —— 设计方案

两层结构：战略层（LLM 解析任务）+ 执行层（AME Frontier覆盖 · 支持禁飞区绕行）

架构总览

战略层 — LLM

  • 人工发布任务，LLM 解析成原子动作
  • 任务明确时解析出 targetName → Patrol("操场", 45)
  • 任务模糊时 targetName 为空 → Patrol("", 45)，由执行层自动选区
→

执行层 — AME DoPatrol()

  • targetName 为空 → 查 Memory 空闲度自动选区
  • 读取特征格子（occupied / approach），过滤禁飞区
  • Frontier-based：始终飞向最近未访问格，遇禁飞区绕行
  • duration 到期或全覆盖后完成，写入巡逻时间戳
→

记录层 — Memory

  • RecordPatrolEvent(name, time) ← AME 调用
  • GetHighestIdlenessArea() ← targetName 为空时 AME 查询
  • BuildPatrolIdlenessContext() ← 空闲度文本（备用）
职责边界：ADM 只负责决策/规划（输出原子动作），不写执行副作用。 巡逻记录属于执行结果，由 AME 在 DoPatrol 完成时写入，与 ADM 解耦。 空闲度上下文不注入 prompt（任务由人工明确发布），仅在 targetName 为空时由执行层查询使用。

Execution/AgentMotionExecutor.cs NEW DoPatrol() 方法（紧接 DoTakeoff 后新增） 
    // ─── Patrol ──────────────────────────────────────────────────

    private IEnumerator DoPatrol(AtomicAction action)
    {
        float totalDuration = action.duration > 0f ? action.duration : 60f;
        bool  withObserve   = action.actionParams?.Contains("observe") == true;

        // ── 0. 目标解析：targetName 为空时查空闲度自动选区 ──────────
        string resolvedTarget = action.targetName;
        if (string.IsNullOrWhiteSpace(resolvedTarget) && _memoryModule != null)
        {
            resolvedTarget = _memoryModule.GetHighestIdlenessArea();
            if (!string.IsNullOrWhiteSpace(resolvedTarget))
                Debug.Log($"[AME] Patrol: targetName 为空，按空闲度选区 → '{resolvedTarget}'");
        }

        // ── 1. 获取巡逻格子集合 ─────────────────────────────────────
        List<Vector2Int> patrolCells = new List<Vector2Int>();

        if (campusGrid != null && !string.IsNullOrWhiteSpace(resolvedTarget))
        {
            // 优先：特征内部格子（适合操场、广场等开放区域）
            // IsWalkable() 自动过滤禁飞区（不可行走格），无需额外处理
            if (campusGrid.TryGetFeatureOccupiedCells(resolvedTarget, out Vector2Int[] occupied))
            {
                patrolCells = occupied
                    .Where(c => campusGrid.IsWalkable(c.x, c.y))
                    .ToList();
            }

            // Fallback：外围接近格（适合建筑物外围巡逻）
            if (patrolCells.Count < 3)
            {
                campusGrid.TryGetFeatureApproachCells(resolvedTarget, transform.position,
                    out Vector2Int[] approach, maxCount: 30);
                patrolCells = approach?.ToList() ?? patrolCells;
            }
        }
        else if (campusGrid != null)
        {
            // targetName 为空且无历史记录：以当前位置为中心取周边 20 格
            Vector2Int center = campusGrid.WorldToGrid(transform.position);
            const int R = 20;
            for (int dx = -R; dx <= R; dx++)
            for (int dz = -R; dz <= R; dz++)
            {
                int x = center.x + dx, z = center.y + dz;
                if (campusGrid.IsInBounds(x, z) && campusGrid.IsWalkable(x, z))
                    patrolCells.Add(new Vector2Int(x, z));
            }
        }

        if (patrolCells.Count == 0) { Debug.LogWarning("[AME] Patrol: 无可用巡逻格，跳过"); yield break; }

        AgentStateServer.PushMotionEvent(props?.AgentID ?? name, "patrol_start",
            $"⬡ 巡逻 '{resolvedTarget}'，{patrolCells.Count} 个格子，时长 {totalDuration}s");

        // ── 2. Frontier-based 覆盖循环 ─────────────────────────────
        HashSet<Vector2Int> visited = new HashSet<Vector2Int>();
        float elapsed = 0f;
        const float VISIT_RADIUS_M  = 8f;
        const float SEG_TIMEOUT     = 15f;
        const float ARRIVE_DIST     = 4f;

        while (elapsed < totalDuration && visited.Count < patrolCells.Count)
        {
            Vector2Int agentCell = campusGrid.WorldToGrid(transform.position);
            Vector2Int? target = null;
            float minDist = float.MaxValue;
            foreach (var c in patrolCells)
            {
                if (visited.Contains(c)) continue;
                float d = Vector2.Distance((Vector2)agentCell, (Vector2)c);
                if (d < minDist) { minDist = d; target = c; }
            }
            if (target == null) break;

            Vector3 goalWorld = campusGrid.GridToWorldCenter(target.Value.x, target.Value.y, hoverHeight);
            float segElapsed = 0f;

            while (segElapsed < SEG_TIMEOUT && elapsed < totalDuration)
            {
                FaceToward(goalWorld);
                ApplyHorizontalPD(goalWorld);

                Vector2 posXZ  = new Vector2(transform.position.x, transform.position.z);
                Vector2 goalXZ = new Vector2(goalWorld.x, goalWorld.z);
                if (Vector2.Distance(posXZ, goalXZ) < ARRIVE_DIST) break;

                segElapsed += Time.deltaTime;
                elapsed    += Time.deltaTime;
                yield return null;
            }

            Vector2Int nowCell = campusGrid.WorldToGrid(transform.position);
            float visitRadiusCells = VISIT_RADIUS_M / campusGrid.cellSize;
            foreach (var c in patrolCells)
            {
                if (Vector2.Distance((Vector2)nowCell, (Vector2)c) <= visitRadiusCells)
                    visited.Add(c);
            }

            if (withObserve) GetComponent<PerceptionModule>()?.SenseOnce();

            float dwell = Random.Range(0.5f, 1.5f);
            float dwellE = 0f;
            Vector3 hold = new Vector3(transform.position.x, hoverHeight, transform.position.z);
            while (dwellE < dwell && elapsed < totalDuration)
            {
                ApplyHorizontalPD(hold);
                dwellE += Time.deltaTime; elapsed += Time.deltaTime;
                yield return null;
            }
        }

        float coverage = patrolCells.Count > 0
            ? (float)visited.Count / patrolCells.Count * 100f : 100f;
        AgentStateServer.PushMotionEvent(props?.AgentID ?? name, "patrol_done",
            $"✓ 巡逻完成 '{resolvedTarget}'，覆盖率 {coverage:F0}%，用时 {elapsed:F0}s");

        // ── 3. 写巡逻时间戳（AME 负责，ADM 不写） ─────────────────
        if (!string.IsNullOrWhiteSpace(resolvedTarget))
            _memoryModule?.RecordPatrolEvent(resolvedTarget, DateTime.Now);
    }
LLM_Modules/MemoryModule.cs NEW 新增巡逻时间戳方法（在类末尾新增） 
    // ─── 巡逻空闲度追踪 ─────────────────────────────────────────────

    private readonly Dictionary<string, DateTime> _patrolTimestamps = new();

    /// DoPatrol 完成时由 AME 调用，记录区域最后巡逻时间
    public void RecordPatrolEvent(string areaName, DateTime time)
    {
        if (!string.IsNullOrWhiteSpace(areaName))
            _patrolTimestamps[areaName] = time;
    }

    /// 返回空闲度最高（最久未巡逻）的区域名；无记录时返回 string.Empty
    /// DoPatrol 在 targetName 为空时调用，自动选区
    public string GetHighestIdlenessArea()
    {
        if (_patrolTimestamps.Count == 0) return string.Empty;
        return _patrolTimestamps
            .OrderByDescending(kv => (DateTime.Now - kv.Value).TotalSeconds)
            .First().Key;
    }

    /// 生成"区域空闲度"文本（备用，供需要时使用）
    public string BuildPatrolIdlenessContext(int topN = 6)
    {
        if (_patrolTimestamps.Count == 0) return "（无历史巡逻记录）";
        var now   = DateTime.Now;
        var lines = _patrolTimestamps
            .OrderByDescending(kv => (now - kv.Value).TotalSeconds)
            .Take(topN)
            .Select(kv =>
            {
                int mins = (int)(now - kv.Value).TotalMinutes;
                string ago = mins == 0 ? "刚巡逻过" : $"{mins}分钟前";
                return $"  • {kv.Key}：{ago}";
            });
        return string.Join("\n", lines);
    }
